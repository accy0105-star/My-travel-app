        const { createApp, ref, computed } = Vue;

        createApp({

            setup() {

                const selectedCity = ref('Reykjavik');

                const currency = ref('ISK');

                const exchangeRate = ref(0.056); // 示例匯率

                const activeDate = ref('2026-01-03');

                const expense = ref({ amount: 0, type: 'card', note: '' });

                const schedule = ref([

                    { date: '2026-01-03', temp: -2, weatherIcon: 'fa-solid fa-snowflake', tasks: [

                        { time: '09:00', location: '凱夫拉維克機場', travelTime: 45 },

                        { time: '11:00', location: '藍湖溫泉 (Blue Lagoon)', travelTime: 60 }

                    ]},

                    { date: '2026-01-04', temp: 1, weatherIcon: 'fa-solid fa-cloud', tasks: [

                        { time: '10:00', location: '黃金圈 (Golden Circle)', travelTime: 30 },

                        { time: '13:00', location: '間歇泉', travelTime: 0 }

                    ]}

                ]);

                const currentDayTasks = computed(() => {

                    const day = schedule.value.find(d => d.date === activeDate.value);

                    return day ? day.tasks : [];

                });

                const currentDayIndex = computed(() => schedule.value.findIndex(d => d.date === activeDate.value));

                const addExpense = () => {

                    alert(`已記錄：${expense.value.amount} ISK (${(expense.value.amount * exchangeRate.value).toFixed(2)} HKD)`);

                    expense.value = { amount: 0, type: 'card', note: '' };

                };

                const handleFileUpload = (e) => { console.log("圖片上傳成功", e.target.files[0]); };

                return { selectedCity, currency, exchangeRate, activeDate, schedule, currentDayTasks, currentDayIndex, expense, addExpense, handleFileUpload };

            }

        }).mount('#app');
